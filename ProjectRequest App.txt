CLASS ZCL_NAD_PROJECT_CODE_REQUEST IMPLEMENTATION.


METHOD /neptune/if_nad_server~handle_on_ajax.

    DATA: lx_ca_form_exception TYPE REF TO zcx_ca_form_exceptions,
          lx_project_request   type ref to zcx_project_code_request,
          ls_neptune_message   TYPE /neptune/message,
          lt_return            TYPE bapiret2_tt,
          ls_return            TYPE bapiret2,
          ls_mail_send         TYPE boole_d,
          lv_mail_to_req       TYPE boole_d.

    CASE ajax_id.

      WHEN 'GET_DATA'.
        TRY.
            CALL METHOD me->get_data
              EXPORTING
                server   = server
                ip_value = ajax_value.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
        ENDTRY.

      WHEN 'GET_EMPINFO'.
        CALL METHOD me->get_userdetail.

      WHEN 'GET_AGENTLIST'.
        try.
          CALL METHOD me->get_agents.

          catch zcx_ca_form_exceptions into lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
        endtry.

      WHEN 'CREATE'.
        TRY.
            CALL METHOD me->create_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
          CATCH zcx_project_code_request into lx_project_request.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_project_request->get_text( ).
        ENDTRY.

      WHEN 'CHECK'.
        TRY.
            CALL METHOD me->check_error
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
        ENDTRY.

      WHEN 'SAVE'.
        TRY.
            CALL METHOD me->save_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
        ENDTRY.

      WHEN 'DELETE'.
        TRY.
            CALL METHOD me->delete_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
        ENDTRY.

      WHEN 'REJECT'.
        TRY.
            CALL METHOD me->reject_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
        ENDTRY.

      WHEN 'SUBMIT'.
        TRY.
          submit_form( ).

          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
          CLEANUP.
        ENDTRY.

      WHEN 'RESUBMIT'.
        TRY.
            CALL METHOD me->resubmit_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
          CLEANUP.
        ENDTRY.

      WHEN 'CHANGE'.
        TRY.
            CALL METHOD me->change_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
          CLEANUP.
        ENDTRY.

      WHEN 'CLOSE'.
        TRY.
            CALL METHOD me->close_form
              EXPORTING
                server = server.
          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exception.
            wa_program_state-program_state   = 'E'.
            wa_program_state-program_message = lx_ca_form_exception->get_text( ).
            wa_program_state-incorrect_field = lx_ca_form_exception->incorrect_field.
          CLEANUP.
        ENDTRY.

        delete_attachment( ).

METHOD /neptune/if_nad_server~handle_on_request.
    DATA: lv_key                TYPE string,
          lv_form_guid          TYPE sysuuid_c,
          lv_attachment_uuid    TYPE string,
          lt_parameters         TYPE string_table,
          lv_parameter          TYPE string,
          lv_param_name         TYPE string,
          lv_value              TYPE string,
          lv_attachment_data    TYPE xstring,
          lv_mime_type          TYPE string,
          lv_file_name          TYPE string,
          lx_ca_form_exceptions TYPE REF TO zcx_ca_form_exceptions.

    lv_key = key.
    TRANSLATE lv_key TO UPPER CASE.

    CASE lv_key.
      WHEN 'ATTACHMENT'.
        IF key_id IS INITIAL.
          redirect_error( server ).
          RETURN.
        ENDIF.

        SPLIT key_id AT ',' INTO TABLE lt_parameters.
        LOOP AT lt_parameters INTO lv_parameter.
          SPLIT lv_parameter AT '=' INTO lv_param_name lv_value.
          TRANSLATE lv_param_name TO UPPER CASE.
          CASE lv_param_name.
            WHEN 'FORM'.
              lv_form_guid = lv_value.
            WHEN 'ATT'.
              lv_attachment_uuid = lv_value.
          ENDCASE.
        ENDLOOP.

        IF lv_form_guid IS INITIAL
            OR lv_attachment_uuid IS INITIAL.
          redirect_error( server ).
          RETURN.
        ENDIF.

        TRY.
            zcl_nd_ca_utilities=>get_attachment_data(
              EXPORTING
                iv_form_guid       = lv_form_guid
                iv_attachment_uuid = lv_attachment_uuid
              IMPORTING
                ev_attachment_data = lv_attachment_data
                ev_mime_type       = lv_mime_type
                ev_file_name       = lv_file_name
            ).

          CATCH zcx_ca_form_exceptions INTO lx_ca_form_exceptions.

        ENDTRY.

        IF lv_attachment_data IS INITIAL.
          redirect_error( server ).
          RETURN.
        ENDIF.

        server->api_response_xstring(
          data         = lv_attachment_data
          content_type = lv_mime_type
          file_name    = lv_file_name
        ).

        RETURN.
      WHEN OTHERS.
    ENDCASE.
  ENDMETHOD.

METHOD agent_search.
  DATA wa_agent_list TYPE ty_agent_list .

  TYPES: BEGIN OF ty_adrp,
           persnumber TYPE adrp-persnumber,
           name_text  TYPE adrp-name_text,
         END OF ty_adrp.

  DATA: it_adrp  TYPE STANDARD TABLE OF ty_adrp,
        it_usr21 TYPE STANDARD TABLE OF usr21,
        wa_usr21 TYPE usr21,
        wa_adrp  TYPE ty_adrp.

  DATA: ra_name TYPE RANGE OF adrp-mc_namefir,
        wa_name LIKE LINE OF ra_name.

  DATA: ls_usr02 TYPE usr02.
*--------------------------------------------------------------------*
* GET DATA
*--------------------------------------------------------------------*

  CHECK ajax_value IS NOT INITIAL.

* LastName
  wa_name-sign   = 'I'.
  wa_name-option = 'CP'.

  SPLIT ajax_value AT space INTO TABLE DATA(lt_name).
  LOOP AT lt_name INTO DATA(lw_name).
    CONCATENATE '*'
                lw_name
                '*'
                INTO wa_name-low.

    TRANSLATE wa_name-low TO UPPER CASE.
    APPEND wa_name TO ra_name.
  ENDLOOP.

* Get ADRP
  SELECT persnumber name_text
         FROM adrp
         INTO TABLE it_adrp
         WHERE mc_namefir IN ra_name
            OR mc_namelas IN ra_name
           AND pers_group IN ( 'BC01', 'BP' ) " Had to include BP for S4.
           AND date_from  LE sy-datum
           AND date_to    GE sy-datum.

* Any hit
  CHECK sy-subrc EQ 0.

* Get USR21
  SELECT *
         FROM usr21
         INTO TABLE it_usr21
         FOR ALL ENTRIES IN it_adrp
         WHERE persnumber EQ it_adrp-persnumber.

* Sorting
  SORT it_usr21 BY persnumber.


*--------------------------------------------------------------------*
* BUILD DATA
*--------------------------------------------------------------------*
  LOOP AT it_adrp INTO wa_adrp.
    READ TABLE it_usr21 INTO wa_usr21 WITH KEY persnumber = wa_adrp-persnumber BINARY SEARCH.

    CHECK sy-subrc EQ 0.

    wa_agent_list-otype = 'U'.
    wa_agent_list-objid = wa_usr21-bname.
    wa_agent_list-text  = wa_adrp-name_text.

*     Check if user is locked or valid dates are no current.
    SELECT SINGLE *
           INTO  ls_usr02
           FROM usr02
           WHERE bname = wa_usr21-bname
             AND gltgv LE sy-datum
             AND gltgb GE sy-datum
             AND uflag IN ( 0 , 128 ).

    IF sy-subrc = 0.

      APPEND wa_agent_list TO gt_agent_list.

    ENDIF.

  ENDLOOP.

* Sorting
  SORT gt_agent_list BY text.
  CLEAR wa_agent_list.

  INSERT wa_agent_list INTO gt_agent_list INDEX 1.

ENDMETHOD.

METHOD create_imo_proj_code.
    DATA lt_zzdci_proj TYPE STANDARD TABLE OF zzdci_proj.
    DATA lv_next_no TYPE fico_nseq_no VALUE '0001'.
    DATA lv_existproj TYPE char4.
    DATA lw_zzdci_proj TYPE zzdci_proj.
    DATA lv_len TYPE i.
    DATA lv_htype TYPE dd01v-datatype.
    DATA lv_year TYPE char2.

* Check whether project code is already created - No duplicates ************************
    IF gs_proj_form-cls_proj_code(3) NE 'IMO'.
* Check End of month to update Year of Project Code request
      IF sy-datum+4(2) EQ '12'.
        lv_year = sy-datum+2(2) + 1.
      ELSE.
        lv_year = sy-datum+2(2).
      ENDIF.

      CONCATENATE 'IMO' lv_year '%' INTO DATA(lw_projcode).
      CONCATENATE 'IMO' lv_year INTO DATA(lw_propose_code).

      SELECT * FROM zzdci_proj INTO TABLE lt_zzdci_proj WHERE zzproject LIKE lw_projcode.
      IF sy-subrc EQ 0.
*        SORT lt_zzdci_proj BY zzproject DESCENDING.
*        LOOP AT lt_zzdci_proj INTO lw_zzdci_proj.
          DATA(lv_count) = lines( lt_zzdci_proj ).
          READ TABLE lt_zzdci_proj INTO lw_zzdci_proj INDEX lv_count.

          lv_len = strlen( lw_zzdci_proj-zzproject ) - 4.
          lv_existproj = lw_zzdci_proj-zzproject+lv_len(4).
          CALL FUNCTION 'NUMERIC_CHECK'
            EXPORTING
              string_in = lv_existproj
            IMPORTING
              htype     = lv_htype.
          IF lv_htype = 'NUMC'.
            IF lv_next_no <= lv_existproj.
              lv_next_no = lv_existproj + 1.
            ENDIF.
          ENDIF.
*        ENDLOOP.
      ENDIF.

      CONCATENATE lw_propose_code lv_next_no INTO gs_proj_form-cls_proj_code SEPARATED BY '_'.

      lw_zzdci_proj-zzproject = gs_proj_form-cls_proj_code.

    ENDIF.

            EXPORTING
              is_order    = ls_proj_form              " MDG - project code request Form

* Update Order detail table
            gs_orderdetails-output_ccode   = ls_proj_form-company_code.
            gs_orderdetails-output_ccenter = ls_proj_form-kostl.

              CALL METHOD deactive_exist_projcode EXPORTING ip_prev_proj_code = ls_prev_proj_code.

              CALL METHOD deactive_exist_projcode EXPORTING ip_prev_proj_code = ls_prev_proj_code.
            ENDIF.

          ELSE.
            DELETE lt_return WHERE type NE 'E'.
            gs_orderdetails-output_ccode   = ls_proj_form-company_code.
            gs_orderdetails-output_ccenter = ls_proj_form-kostl.
            gs_orderdetails-project_code   = ls_proj_form-project_code.

              LOOP AT lt_return INTO lw_return WHERE type = 'E'.
                IF sy-tabix EQ 1.
                  lv_message = lw_return-message.
                ELSE.
                  lv_message = | { '/' } { lv_message } { lw_return-message }|.
                ENDIF.
              ENDLOOP.
              gs_orderdetails-is_error       = gc_true.
              gs_orderdetails-error_desc = ls_proj_form-internal_no_err = lv_message.

                CALL METHOD deactive_exist_projcode EXPORTING ip_prev_proj_code = ls_prev_proj_code.

            ENDIF.
          ENDIF.
        ENDIF.

        APPEND gs_orderdetails TO gt_orderdetails.
        MOVE-CORRESPONDING ls_proj_form TO gs_proj_form.
      ELSE.
* Update Order detail table
        gs_orderdetails-output_ccode   = ls_proj_form-company_code.
        gs_orderdetails-output_ccenter = ls_proj_form-kostl.
        gs_orderdetails-internal_order = ls_proj_form-internal_no.

    IF gs_proj_form-project_type IS NOT INITIAL.

METHOD get_proj_code.
    DATA lt_zzdci_proj TYPE STANDARD TABLE OF zzdci_proj.
    DATA lv_next_no TYPE fico_nseq_no VALUE '0001'.
    DATA lv_existproj TYPE char4.
    DATA lw_zzdci_proj TYPE zzdci_proj.
    DATA lv_len TYPE i.

    DATA lv_year TYPE char2.

    IF gs_proj_form-project_type IS NOT INITIAL.
      IF sy-datum+4(2) EQ '12'.
        lv_year = sy-datum+2(2) + 1.
      ELSE.
        lv_year = sy-datum+2(2).
      ENDIF.

      CASE gs_proj_form-project_type.
        WHEN '1' OR '4'.
          CONCATENATE 'CAP' lv_year '%' INTO DATA(lw_projcode).
          CONCATENATE 'CAP' lv_year INTO DATA(lw_propose_code).
        WHEN '2'.
          CONCATENATE 'ITL' lv_year '%' INTO lw_projcode.
          CONCATENATE 'ITL' lv_year INTO lw_propose_code.
        WHEN '3'.
          CONCATENATE 'CLS' lv_year '%' INTO lw_projcode.
          CONCATENATE 'CLS' lv_year INTO lw_propose_code.

          CONCATENATE 'IMO' lv_year '%' INTO lw_projcode.
          CONCATENATE 'IMO' lv_year INTO lw_propose_code.

      IF lw_propose_code(3) NE gs_proj_form-project_code(3).
* Read Existing Project Code
        SELECT * FROM zzdci_proj INTO TABLE lt_zzdci_proj WHERE zzproject LIKE lw_projcode.
        IF sy-subrc EQ 0.
          DATA(lv_count) = lines( lt_zzdci_proj ).

          READ TABLE lt_zzdci_proj INTO lw_zzdci_proj INDEX lv_count.

*          SORT lt_zzdci_proj BY zzproject DESCENDING.
*          LOOP AT lt_zzdci_proj INTO lw_zzdci_proj.

          lv_len = strlen( lw_zzdci_proj-zzproject ) - 4.
          lv_existproj = lw_zzdci_proj-zzproject+lv_len(4).

          CALL FUNCTION 'NUMERIC_CHECK'
            EXPORTING
              string_in = lv_existproj
            IMPORTING
              htype     = lv_htype.
          IF lv_htype = 'NUMC'.
            IF lv_next_no <= lv_existproj.
              lv_next_no = lv_existproj + 1.
            ENDIF.
          ENDIF.

*          ENDLOOP.
        ENDIF.

        CONCATENATE lw_propose_code lv_next_no INTO gs_proj_form-project_code SEPARATED BY '_'.

* Update Project Code
        lw_zzdci_proj-zzproject = gs_proj_form-project_code.
        lw_zzdci_proj-zzdescription = |{ gs_proj_form-project_code } { gs_proj_form-projectname }|.

        IF gs_proj_form-project_type EQ '4'  AND gs_proj_form-order_type EQ 'Z060'.
          lw_zzdci_proj-cap_active = ''.
        ELSEIF ( gs_proj_form-project_type EQ '1' OR gs_proj_form-project_type EQ '4' OR gs_proj_form-project_type EQ '5' ) AND gs_proj_form-order_type NE 'Z060'.
          lw_zzdci_proj-cap_active = 'X'.
        ELSEIF gs_proj_form-project_type EQ '2' AND gs_proj_form-order_type NE 'Z060'.
          lw_zzdci_proj-labor_ind = 'X'.
        ENDIF.

METHOD save_attachments.
    DATA: lr_utilities  TYPE REF TO zcl_nd_ca_utilities,
          lv_form_guid  TYPE sysuuid_c,
          lv_file_data  TYPE xstring,
          lv_error      TYPE flag,
          ls_zmdgattach TYPE zmdgattach.


    FIELD-SYMBOLS: <attachment> TYPE zca_form_attachment.

    lv_form_guid = gs_formdetails-form_guid.

    CREATE OBJECT lr_utilities.

    LOOP AT gt_attachments ASSIGNING <attachment>
        WHERE uuid IS INITIAL.
      CALL FUNCTION 'SCMS_BASE64_DECODE_STR'
        EXPORTING
          input  = <attachment>-file_data
        IMPORTING
          output = lv_file_data
        EXCEPTIONS
          OTHERS = 1.

      <attachment>-uuid = lr_utilities->add_attachment_to_form(
        iv_form_guid   = lv_form_guid
        iv_doc_name    = <attachment>-file_name
        iv_doc_content = lv_file_data
      ).

      IF <attachment>-uuid IS INITIAL.
        lv_error = 'X'.
        DELETE gt_attachments.
        CONTINUE.
      ELSE.
*     Don't pass attachment data back in the response.
        CLEAR <attachment>-file_data.
*     Save categorization
        CLEAR ls_zmdgattach.
        ls_zmdgattach-form_guid     = lv_form_guid.
        ls_zmdgattach-attachment_id = <attachment>-uuid.
        ls_zmdgattach-category      = <attachment>-category.
        MODIFY zmdgattach FROM ls_zmdgattach.
      ENDIF.
    ENDLOOP.

    IF NOT lv_error IS INITIAL.
*   Some attachments could not be uploaded.
      RAISE EXCEPTION TYPE zcx_ca_form_exceptions
        EXPORTING
          textid = zcx_ca_form_exceptions=>attachment_failure.
      RETURN.
    ENDIF.
  ENDMETHOD.

METHOD search_prev_cap_proj_id.
    DATA wa_agent_list TYPE ty_agent_list .

    TYPES: BEGIN OF ty_adrp,
             persnumber TYPE adrp-persnumber,
             name_text  TYPE adrp-name_text,
           END OF ty_adrp.

    DATA: it_adrp  TYPE STANDARD TABLE OF ty_adrp,
          it_usr21 TYPE STANDARD TABLE OF usr21,
          wa_usr21 TYPE usr21,
          wa_adrp  TYPE ty_adrp.

    DATA: ra_name TYPE RANGE OF adrp-mc_namefir,
          wa_name LIKE LINE OF ra_name.


*--------------------------------------------------------------------*
* GET DATA
*--------------------------------------------------------------------*

    CHECK ajax_value IS NOT INITIAL.

* LastName
    wa_name-sign   = 'I'.
    wa_name-option = 'CP'.

    CONCATENATE '*'
                ajax_value
                '*'
                INTO wa_name-low.

    TRANSLATE wa_name-low TO UPPER CASE.
    APPEND wa_name TO ra_name.

* Get ADRP
    SELECT persnumber name_text
           FROM adrp
           INTO TABLE it_adrp
           WHERE mc_namefir IN ra_name
              OR mc_namelas IN ra_name
             AND pers_group EQ 'BC01'
             AND date_from  LE sy-datum
             AND date_to    GE sy-datum.

* Any hit
    CHECK sy-subrc EQ 0.

* Get USR21
    SELECT *
           FROM usr21
           INTO TABLE it_usr21
           FOR ALL ENTRIES IN it_adrp
           WHERE persnumber EQ it_adrp-persnumber.

* Sorting
    SORT it_usr21 BY persnumber.


*--------------------------------------------------------------------*
* BUILD DATA
*--------------------------------------------------------------------*
    LOOP AT it_adrp INTO wa_adrp.
      READ TABLE it_usr21 INTO wa_usr21 WITH KEY persnumber = wa_adrp-persnumber BINARY SEARCH.

      CHECK sy-subrc EQ 0.

      wa_agent_list-otype = 'U'.
      wa_agent_list-objid = wa_usr21-bname.
      wa_agent_list-text  = wa_adrp-name_text.
      APPEND wa_agent_list TO gt_agent_list.
    ENDLOOP.

* Sorting
    SORT gt_agent_list BY text.
    CLEAR wa_agent_list.

    INSERT wa_agent_list INTO gt_agent_list INDEX 1.

  ENDMETHOD.

METHOD set_button_visibility.
    DATA lt_dfies TYPE dfies_tab.
    TYPES : BEGIN OF ty_agentid,
              sign   TYPE char1,
              option TYPE char2,
              low    TYPE zform_appragent,
              high   TYPE zform_appragent.
    TYPES : END OF ty_agentid.

    DATA lv_myrequest   TYPE boole_d.
    DATA lv_initiate_by TYPE syuname.

* Step 1
* Disable All button and enable based on role and approval step
    CALL FUNCTION 'DDIF_FIELDINFO_GET'
      EXPORTING
        tabname   = 'ZMDG_S_NAD_ORDER_BUTTONS'
      TABLES
        dfies_tab = lt_dfies.

    LOOP AT lt_dfies INTO DATA(lw_dfies) WHERE fieldname NE 'FORM_GUID'.
      ASSIGN COMPONENT lw_dfies-fieldname OF STRUCTURE gs_proj_form TO FIELD-SYMBOL(<fs>).
      <fs> = gc_false.
    ENDLOOP.

* Check whether form is currently processed by other user
    CHECK is_form_locked IS NOT INITIAL.

    if is_form_locked = 'X'.
      gs_proj_form-can_attach = 'X'.
    endif.

* Check Form Status
    IF ( gs_formdetails-form_status EQ gc_formstatus_completed OR gs_formdetails-form_status EQ gc_formstatus_cancelled ) AND wa_url_parameters-mode NE 'EDIT'.
      EXIT.
    ENDIF.

* Update Form Intiated by
    IF gs_proj_form-requester_id IS NOT INITIAL.
      lv_initiate_by = gs_proj_form-requester_id.
    ELSE.
      lv_initiate_by = gs_formdetails-initiated_by.
    ENDIF.

* Read Approver List
    SELECT * FROM zca_apprvr_list INTO TABLE @DATA(lt_apprvr_list) WHERE form_name = @gc_form_name AND form_ver = @gc_form_ver.


    IF gs_formdetails-form_guid IS NOT INITIAL.
      gs_proj_form-form_guid = gs_formdetails-form_guid.
      DATA(lw_apprvr_list) = lt_apprvr_list[ appr_step = gs_formdetails-appr_step ].

      IF lv_initiate_by EQ sy-uname AND lw_apprvr_list-appr_seqno EQ '0001'.
        lv_myrequest = 'X'.
      ELSE.
        lv_myrequest = ''.
      ENDIF.
    ELSE.
      lv_myrequest = 'X'.
    ENDIF.

* #1 Check Authorization for WITHDRAWN/Close Button**********************************************************************
    IF gs_formdetails-form_status EQ gc_formstatus_rejected AND lv_initiate_by EQ sy-uname AND wa_url_parameters-mode NE 'DISPLAY' AND
               ( gs_formdetails-form_status = 'REJECTED' AND gs_proj_form-request_from NE 'U' AND wa_url_parameters-from NE 'search').

      EXIT.
    ELSEIF gs_formdetails-form_status EQ gc_formstatus_rejected AND lv_initiate_by EQ sy-uname AND wa_url_parameters-mode NE 'DISPLAY' AND

      gs_proj_form-withdrawn = gc_true.
      EXIT.
    ELSEIF ( gs_formdetails-form_status EQ gc_formstatus_rejected AND lv_initiate_by NE sy-uname AND wa_url_parameters-mode NE 'EDIT' ) OR wa_url_parameters-mode EQ 'DISPLAY'.
      EXIT.

METHOD set_header.
    DATA: lv_date TYPE sy-datum.
    DATA: lv_time TYPE sy-uzeit.
    DATA: lv_date_c(10) TYPE c.
    DATA: lv_time_c(10) TYPE c.
    DATA: lw_agent_list TYPE zca_s_nad_gl_agent.

    IF gs_formdetails-form_guid IS NOT INITIAL AND NOT ( wa_url_parameters-mode EQ 'EDIT' AND wa_url_parameters-from = 'search' AND gs_formdetails-appr_step = 'BPC' ).
      MOVE-CORRESPONDING gs_formdetails TO gs_formheader.
      MOVE gs_formdetails-latest_startdate    TO lv_date.
      MOVE gs_formdetails-latest_starttime    TO lv_time.
      gs_formdetails-initiated_by = gs_proj_form-requester_id.
    ELSE.
      gs_formdetails-initiated_by = sy-uname.
      lv_date                     = sy-datum.
      lv_time                     = sy-uzeit.
      gs_formheader-current_step  = 'Unsaved Draft'.
    ENDIF.

    WRITE lv_date     TO lv_date_c.                    " Format Date and Time
    WRITE lv_time     TO lv_time_c.
    MOVE  lv_date_c   TO gs_formheader-start_date.
    MOVE  lv_time_c   TO gs_formheader-start_time.

*Object Name**********************************************************
    IF gs_proj_form-projectname IS NOT INITIAL.
      gs_formheader-object_name = gs_proj_form-projectname.
    ELSE.
      gs_formheader-object_name = 'New project code request'.
    ENDIF.

*Get Approver Step Detail*********************************************************
    IF gs_formdetails-appr_step IS NOT INITIAL.
      SELECT SINGLE * FROM zca_apprvr_list INTO @DATA(lw_apprvr_list) WHERE form_name = @gc_form_name AND form_ver = @gc_form_ver AND appr_step  = @gs_formdetails-appr_step.
      IF sy-subrc EQ 0.
        gs_formheader-current_step = lw_apprvr_list-appr_desc.
      ENDIF.
    ENDIF.

*Current Agent List***************************************************
    IF gs_formdetails-agent_id_list IS NOT INITIAL.
      CLEAR lw_apprvr_list-workcenter_id.
    ENDIF.

    gt_zca_agent_list = zcl_nd_ca_mdg_workflow=>get_agents_of_approval_step(
      iv_form_guid     = gs_formdetails-form_guid
      iv_agent_id_list = gs_formdetails-agent_id_list
      iv_work_center   = lw_apprvr_list-workcenter_id
    ).

    LOOP AT gt_zca_agent_list INTO lw_agent_list.
      CASE sy-tabix.
        WHEN 1.
          gs_formheader-current_agent = lw_agent_list-agent_name.
        WHEN 2.
          CONCATENATE gs_formheader-current_agent lw_agent_list-agent_name INTO gs_formheader-current_agent SEPARATED BY ' / '.
        WHEN OTHERS.
          CONCATENATE gs_formheader-current_agent 'more' INTO gs_formheader-current_agent SEPARATED BY '...'.
          EXIT.
      ENDCASE.
    ENDLOOP.

*Requester ID*********************************************************
    CALL METHOD me->get_userdetail.
    gs_formheader-initiated_by = gs_employee_info-ename.

*Check Current Agent**************************************************
    IF gs_formheader-current_agent IS INITIAL.
      gs_formheader-current_agent = gs_employee_info-ename.
    ENDIF.

    IF gs_formdetails-form_status = gc_formstatus_completed.
      gs_formheader-current_step = 'Complete'.
      CLEAR gs_formheader-current_agent.
    ENDIF.

    IF gs_formdetails-form_status = gc_formstatus_cancelled.
      gs_formheader-current_step = 'Withdrawn'.
      CLEAR gs_formheader-current_agent.
    ENDIF.

    IF wa_url_parameters-mode EQ 'EDIT' AND wa_url_parameters-from = 'search' AND gs_formdetails-appr_step = 'BPC' .
      gs_formheader-current_step  = 'Unsaved Draft'.
      gs_formheader-current_agent = gs_employee_info-ename.
    ENDIF.

  ENDMETHOD.


METHOD set_multi_ccode.

    DATA: lt_ccodes TYPE TABLE OF  zmdg_prj_ccode.
    DATA: lw_ccode  TYPE zmdg_prj_ccode.

* Get list of company code - To include region in the multiple company code
    SELECT * INTO TABLE lt_ccodes FROM zmdg_prj_ccode.
    IF sy-subrc NE 0.
      REFRESH lt_ccodes.
    ENDIF.

* Populate Multiple Company Code
    SPLIT gs_proj_form-additional_comp_code AT ',' INTO TABLE DATA(lt_compcode).
    SPLIT gs_proj_form-more_addi_compcode   AT ',' INTO TABLE DATA(lt_more_compcode).

    LOOP AT lt_compcode INTO DATA(lw_multicompcode).
      SPLIT lw_multicompcode AT '/' INTO DATA(lv_companycode)
                                         DATA(lv_costcenter)
                                         DATA(lv_projcode)
                                         DATA(lv_internalorderno)
                                         DATA(lv_itl_projcode)
                                         DATA(lv_itl_order)
                                         DATA(lv_status)
                                         DATA(lv_cls_projcode)
                                         DATA(lv_cls_order).

* Get Company Code Text
      CONDENSE lv_companycode.
      CONDENSE lv_costcenter.

      CONCATENATE lv_companycode lv_costcenter INTO DATA(lv_concate_ccode) SEPARATED BY '/' .

      READ TABLE gt_all_comp_code INTO gs_comp_code WITH KEY value = lv_companycode.
      IF sy-subrc EQ 0.
        gs_multipleccode-comp_code = lv_companycode.
        gs_multipleccode-comp_text = gs_comp_code-text.
      ENDIF.

* Get Company Code Text
      IF lv_costcenter IS NOT INITIAL.
        SELECT SINGLE * FROM cskt INTO @DATA(lw_cskt) WHERE spras = 'E' AND kostl EQ @lv_costcenter.
        IF sy-subrc EQ 0.
          gs_multipleccode-kostl_code = lv_costcenter.
          gs_multipleccode-kostl_text = lw_cskt-ltext.
        ENDIF.
      ENDIF.

* Get Region
      READ TABLE lt_ccodes INTO lw_ccode WITH KEY company_code = lv_companycode.
      IF sy-subrc EQ 0.
        CASE lw_ccode-processing_region.
          WHEN 'U'.
            gs_multipleccode-region = 'US/LA'.
          WHEN 'E'.
            gs_multipleccode-region = 'Europe'.
          WHEN 'A'.
            gs_multipleccode-region = 'Asia'.
        ENDCASE.
      ELSE.
        CLEAR gs_multipleccode-region .
      ENDIF.

* Set delete function only when Order didn't get created
      IF lv_internalorderno IS INITIAL AND lv_itl_order IS INITIAL AND lv_cls_order IS INITIAL AND
gs_formdetails-appr_step = 'BUDGET_MANAGER' AND gs_formdetails-form_status NE gc_formstatus_completed AND wa_url_parameters-mode NE 'DISPLAY'.
        gs_multipleccode-deletable = gc_true.
      ELSE.
        gs_multipleccode-deletable = gc_false.
      ENDIF.

* Update Status ICON
      READ TABLE lt_more_compcode INTO DATA(lw_more_compcode) WITH KEY table_line = lv_concate_ccode.
      IF sy-subrc EQ 0.
        gs_multipleccode-status    = 'sap-icon://document'.
        DELETE lt_more_compcode INDEX sy-tabix.
      ELSE.
        gs_multipleccode-status    = ''.
      ENDIF.

      APPEND gs_multipleccode TO gt_additional_ccode.
    ENDLOOP.

    IF lv_projcode IS NOT INITIAL.
      DELETE ADJACENT DUPLICATES FROM gt_additional_ccode COMPARING comp_code kostl_code.
    ENDIF.

* Include More Additional company code
    LOOP AT lt_more_compcode INTO lw_multicompcode.
      SPLIT lw_multicompcode AT '/' INTO lv_companycode
                                         lv_costcenter.

* Get Company Code Text
      CONDENSE lv_companycode.
      CONDENSE lv_costcenter.

      READ TABLE gt_all_comp_code INTO gs_comp_code WITH KEY value = lv_companycode.
      IF sy-subrc EQ 0.
        gs_multipleccode-comp_code = lv_companycode.
        gs_multipleccode-comp_text = gs_comp_code-text.
      ENDIF.

* Get Company Code Text
      IF lv_costcenter IS NOT INITIAL.
        SELECT SINGLE * FROM cskt INTO lw_cskt WHERE spras = 'E' AND kostl EQ lv_costcenter.
        IF sy-subrc EQ 0.
          gs_multipleccode-kostl_code = lv_costcenter.
          gs_multipleccode-kostl_text = lw_cskt-ltext.
        ENDIF.
      ENDIF.

* Set delete function only when Order didn't get created
      IF lv_internalorderno IS INITIAL AND lv_itl_order IS INITIAL AND lv_cls_order IS INITIAL AND
gs_formdetails-appr_step = 'BUDGET_MANAGER' AND gs_formdetails-form_status NE gc_formstatus_completed AND wa_url_parameters-mode NE 'DISPLAY'.
        gs_multipleccode-deletable = gc_true.
      ELSE.
        gs_multipleccode-deletable = gc_false.
      ENDIF.

* Update Status ICON
      gs_multipleccode-status    = 'sap-icon://document'.

* Get Region
      READ TABLE lt_ccodes INTO lw_ccode WITH KEY company_code = lv_companycode.
      IF sy-subrc EQ 0.
        CASE lw_ccode-processing_region.
          WHEN 'U'.
            gs_multipleccode-region = 'US/LA'.
          WHEN 'E'.
            gs_multipleccode-region = 'Europe'.
          WHEN 'A'.
            gs_multipleccode-region = 'Asia'.
        ENDCASE.
      ELSE.
        CLEAR gs_multipleccode-region .
      ENDIF.

      APPEND gs_multipleccode TO gt_additional_ccode.
    ENDLOOP.

  ENDMETHOD.

method CC_ALL_APPROVERS.
  data: lt_workcenters  type table of hrobjid,
        lv_workcenter   type hrobjid,
        lv_form_guid    type sysuuid_c,
        lv_form_version type zform_ver,
        lt_users        type zca_t_nad_gl_agent,
        ls_agent        type zca_s_nad_gl_agent,
        lv_user_id      type xubname.

  CASE iv_processing_region.
    WHEN 'E'.
      lv_form_version = '0002'.
    WHEN 'A'.
      lv_form_version = '0003'.
    WHEN OTHERS.
      lv_form_version = '0001'.
  ENDCASE.

  select workcenter_id into table lt_workcenters
      from zca_apprvr_list
      where form_name = iv_form_name
        and form_ver = lv_form_version
        and workcenter_id > 0
        and appr_seqno > 1.

  loop at lt_workcenters into lv_workcenter.
    refresh lt_users.
    try.
      lt_users = zcl_nd_ca_mdg_workflow=>get_agents_of_approval_step(
        iv_form_guid = lv_form_guid
        iv_work_center = lv_workcenter
      ).

      catch zcx_ca_form_exceptions.
        continue.
    endtry.

    append lines of lt_users to rt_agents.
  endloop.

* Add the on-behalf user, if they were filled in.
  if not iv_onbehalf_of is initial.
    lv_user_id = iv_onbehalf_of.

    clear ls_agent.
    ls_agent-agent_id = iv_onbehalf_of.
    ls_agent-agent_name = zcl_nd_ca_utilities=>get_name_from_user( lv_user_id ).
    append ls_agent to rt_agents.
  endif.

  sort rt_agents by agent_id.
  delete adjacent duplicates from rt_agents comparing agent_id.

  export AGENTS from rt_agents  to memory id 'ZMDG_CC'.
endmethod.


method CREATE_PROJECT.
  data: ls_project type zps_project_update,
        lv_template type ps_vorspsp,
        lv_subtemplate_id type zmdg_proj_subtype,
        lt_return type table of bapiret2,
        ls_return type bapiret2,
        ls_zmdg_proj_form type zmdg_proj_form,
        lv_project_subtype type zmdg_proj_subtype,
        lv_budget type flag.

  ls_project-description = gs_proj_form-projectname.
  ls_project-controlling_area = '1000'.
  ls_project-company_code = gs_proj_form-company_code.
  ls_project-completion_date = gs_proj_form-proj_completion_date.
  ls_project-cost_center = gs_proj_form-kostl.
  ls_project-profit_center = gs_proj_form-profit_center.
  ls_project-cap_id = gs_proj_form-capid.
  ls_project-brand = gs_proj_form-brand.
  ls_project-geography = gs_proj_form-geography.
  ls_project-budget = gs_proj_form-capex_req_amount.
  ls_project-budget_currency = gs_proj_form-capex_req_curr.
  ls_project-location = gs_proj_form-location.

  if not gs_proj_form-resp_lead is initial.
*   Read existing VERNR from TCJ04, insert new record if not found.
    ls_project-person_responsible = zcl_nd_project_utility=>insert_tcj04( gs_proj_form-resp_lead ).
  endif.

* If a project was already created, attempt to update it.
  ls_project-project = gs_proj_form-project_id.

  if gs_proj_form-project_type = '5'.
    lv_project_subtype = gs_proj_form-project_subtype.
  else.
    lv_project_subtype = '00'.
  endif.

  if gs_proj_form-project_type = '6'.
    lv_budget = ''.
  else.
    lv_budget = 'X'.
  endif.

  select single standard_wbs into lv_template
      from zps_template_map
      where project_type = gs_proj_form-project_type
        and project_subtype = lv_project_subtype.

  if sy-subrc <> 0.
    raise exception type zcx_project_code_request
      exporting
        textid = zcx_project_code_request=>project_type_invalid
        project_type = gs_proj_form-project_type.
  endif.

  call function 'Z_PS_COPY_TEMPLATE'
    exporting
      iv_template = lv_template
      is_project = ls_project
      iv_update_budget = lv_budget
      iv_settlement = lv_budget
    importing
      ev_project = gs_proj_form-project_id
    tables
      return = lt_return.

  read table lt_return with key type = 'E' transporting no fields.
  if sy-subrc = 0 or gs_proj_form-project_id is initial.
    send_error_email( ).

*   Project creation either didn't work out, or there were issues releasing the project.
    if not gs_proj_form-project_id is initial.
*     In this case, the project was created ok, but it either wasn't updated or wasn't released.
*     The form needs to be updated with the project id.
      clear ls_zmdg_proj_form.
      move-corresponding gs_proj_form to ls_zmdg_proj_form.
      modify zmdg_proj_form from ls_zmdg_proj_form.
      commit work.

      raise exception type zcx_project_code_request
        exporting
          textid = zcx_project_code_request=>created_but_not_released
          project_id = gs_proj_form-project_id
          messages = lt_return.
    else.
*     In this case, the project was not created at all.
      raise exception type zcx_project_code_request
        exporting
          textid = zcx_project_code_request=>error_during_creation
          messages = lt_return.
    endif.
  else.
*   Project was created and released.
    cc_all_approvers(
      iv_form_name = gc_form_name
      iv_processing_region = gs_proj_form-processing_region
      iv_onbehalf_of = gs_proj_form-onbehalf_id
    ).
    submit_form( ).
  endif.

  if not gs_proj_form-project_id is initial.
    call function 'CONVERSION_EXIT_PROJN_OUTPUT'
      exporting
        input  = gs_proj_form-project_id
      importing
        output = gs_proj_form-project_id_out.
  endif.
endmethod.

method VALIDATE_BRAND_GEO.
  select count( * ) from ztrr_cc_brand
      where bukrs = gs_proj_form-company_code
        and brand = gs_proj_form-brand.
  if sy-subrc <> 0.
    raise exception type zcx_project_code_request
      exporting
        textid = zcx_project_code_request=>brand_not_valid
        brand = gs_proj_form-brand
        company_code = gs_proj_form-company_code.
  endif.

  select count( * ) from ztrr_cc_geo
      where bukrs = gs_proj_form-company_code
        and geography = gs_proj_form-geography.
  if sy-subrc <> 0.
    raise exception type zcx_project_code_request
      exporting
        textid = zcx_project_code_request=>geography_not_valid
        geography = gs_proj_form-geography
        company_code = gs_proj_form-company_code.
  endif.
endmethod.


method CURRENCY_CONVERT_IN.
  data: lv_amount type dec11_4.

  try.
    lv_amount = cv_amount.

    call function 'CURRENCY_AMOUNT_DISPLAY_TO_SAP'
      exporting
        currency = iv_currency
        amount_display = lv_amount
      importing
        amount_internal = lv_amount
      exceptions
        others = 1.

    cv_amount = lv_amount.

    catch CX_SY_CONVERSION_NO_NUMBER.
      cv_amount = 0.
  endtry.
endmethod.


method CURRENCY_CONVERT_OUT.
  data: lv_amount type dec11_4.

  try.
    lv_amount = cv_amount.

    call function 'CURRENCY_AMOUNT_SAP_TO_DISPLAY'
      exporting
        currency = iv_currency
        amount_internal = lv_amount
      importing
        amount_display = lv_amount
      exceptions
        others = 1.

    cv_amount = lv_amount.

    catch CX_SY_CONVERSION_NO_NUMBER.
      cv_amount = 0.
  endtry.
endmethod.
