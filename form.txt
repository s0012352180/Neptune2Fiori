                    field.focus();

                }

        });
    });

    return error;
}

function validateRequired(input, form) {
    ////////// ;
    var error = false;
    if (typeof input.setValueState === "function") {
        input.setValueState(sap.ui.core.ValueState.None);
    }

    if (!input.getVisible()) return false;
    if (typeof input.getEnabled === "function" && !input.getEnabled()) return false;

    if (input instanceof sap.m.Input || input instanceof sap.m.TextArea || input instanceof sap.m.DatePicker) {
        if (input.getValue().length === 0) {
            // autoExpand(form);
            input.setValueState(sap.ui.core.ValueState.Error);
            input.setValueStateText("Entry is required.");
            input.setShowValueStateMessage(true);
            input.focus();
            error = true;
        }
    } else if (input instanceof sap.m.Switch) {
        if (!input.getState()) {
            input.addStyleClass("error-switch");
            if (input === oSwitchFPAAppr ||
                input === oSwitchRTRAppr ||
                input === oSwitchBPCAppr) {
                discoveryFiori.showMessage("Select 'YES' to acknowledge before you submit the form", sap.ui.core.ValueState.Error);
                error = true;
            }
        }
    } else if (input instanceof sap.m.Select || input instanceof sap.m.MultiComboBox) {
        if (input.getSelectedKey().length === 0) {
            input.setValueState(sap.ui.core.ValueState.Error);
            input.setValueStateText("Entry is required.");
            input.focus();
            error = true;
        }
    } else if (input instanceof sap.m.RadioButtonGroup) {

        error = false;
        var selected = false;
        input.getButtons().forEach(function(btn) {
            if (btn.getSelected()) selected = true;
        });

        if (!selected) {
            input.setValueState(sap.ui.core.ValueState.Error);
            input.focus();
            error = true;

        }

    } else if (input instanceof sap.m.VBox || input instanceof sap.m.HBox) {
        input.getItems().forEach(function(item) {
            if (validateRequired(item, form)) error = true;
        });
    }

    return error;
}

function validateAttachment() {
    var error = false;
    var count = 0;

    var type = modeloPageProj.oData.PROJECT_TYPE;
    var subtype = (typeof modeloPageProj.oData.PROJECT_SUBTYPE === "string" && modeloPageProj.oData.PROJECT_SUBTYPE.trim().length > 0)
        ? modeloPageProj.oData.PROJECT_SUBTYPE : "00";

    var required = ModelData.Find(AttachmentsRequired, ["PROJECT_TYPE","PROJECT_SUBTYPE","REQUIRED"],
        [type,subtype,true],
        ["EQ","EQ","EQ"]);

    required.forEach(function(category) {
        if (type === "3" && category.ATT_CATEGORY === "ACCOUNTING" && !rbITLYes.getSelected()) {
            return;
        }

        if (ModelData.Find(listAttachments, "CATEGORY", category.ATT_CATEGORY, "EQ").length === 0) {
            var cat = ModelData.Find(AttachmentCategories,"VALUE",category.ATT_CATEGORY,"EQ");
            if (cat.length) {
                if (type !== "3") {
                    discoveryFiori.showMessage("You must attach " + cat[0].TEXT + " to this form before you submit it for approval.", sap.ui.core.ValueState.Error);
                }
                error = true;
            }
        }
        else {
            // Track how many requirements were satisfied.
            count++;
        }
    });

    if (type === "3") {
        if (count > 1) {
            error = false;
        }
        else {
            if (rbITLYes.getSelected()) {
                discoveryFiori.showMessage("You must attach the FP&A Approval Document and either"
                + " the Accounting Questionnaire or CAPEX Software License"
                + " to this form before you submit it for approval.", sap.ui.core.ValueState.Error);
            }
            else {
                discoveryFiori.showMessage("You must attach the FP&A Approval Document and"
                + " the CAPEX Software License"
                + " to this form before you submit it for approval.", sap.ui.core.ValueState.Error);
            }
        }
    }

    return error;
}

function validateAmount() {
    var error = false;

    if (modeloPageProj.oData.PROJECT_TYPE === "6") {
        return;
    }

    var capamount = Number(inpCAPEX_Amount.getValue());
    var itlamount = Number(inpITL_Amount.getValue());

    if ((oLblCAPEX_AMOUNT.getRequired()) && capamount <= 0) {
        inpCAPEX_Amount.setValueState(sap.ui.core.ValueState.Error);
        inpCAPEX_Amount.setValueStateText("The CAPEX Amount should be greater 0");
        inpCAPEX_Amount.setShowValueStateMessage(true);
        inpCAPEX_Amount.focus();
        return true;
    } else if ((oLblITL_AMOUNT.getRequired()) && itlamount <= 0) {
        inpITL_Amount.setValueState(sap.ui.core.ValueState.Error);
        inpITL_Amount.setValueStateText("The ITL requested Amount should be greater 0");
        inpITL_Amount.setShowValueStateMessage(true);
        inpITL_Amount.focus();
        return true;
    } else if ((oLblCAPEX_AMOUNT.getRequired()) && oSelectCAPCURR.getSelectedKey().length === 0) {
        oSelectCAPCURR.setValueState(sap.ui.core.ValueState.Error);
        oSelectCAPCURR.setValueStateText("Currency is required");
        oSelectCAPCURR.focus();
        error = true;
    } else if ((oLblITL_AMOUNT.getRequired()) && oSelectITLCURR.getSelectedKey().length === 0) {
        oSelectITLCURR.setValueState(sap.ui.core.ValueState.Error);
        oSelectITLCURR.setValueStateText("Currency is required");
        oSelectITLCURR.focus();
        error = true;
    }
}

function resetFields() {
    forms.forEach(function(form) {
        form.getAggregation("form").getAggregation("formContainers")[0].getAggregation("formElements").forEach(function(element) {
            var field = element.getFields()[0];
            if (field instanceof sap.m.Input || field instanceof sap.m.Select) {
                field.setValueState(sap.ui.core.ValueState.None);
            } else if (field instanceof sap.m.Switch) {
                field.removeStyleClass("error-switch");
            }
        });
    });
}
